jQuery.domsearch plugin example

JavaScript:

$(document).ready(function () {
    $('#search-field').domsearch({
        searchIn: '.content-item',
        highlight: true
    });
});

CSS:

.highlight {
    font-weight:bold;
    background-color:#DDDDDD;
}
